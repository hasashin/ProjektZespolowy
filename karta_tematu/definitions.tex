\newcommand{\readFileContents}[1]{
\newread\file
\openin\file=#1
\ifeof\file
\errmessage{There is no file #1}
\else
\def\inputtext{}
\loop\unless\ifeof\file
    \read\file to\line
    \edef\inputtext{\inputtext\line}
\repeat
\closein\file
\fi
}

}
